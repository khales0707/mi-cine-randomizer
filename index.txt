    <script>
        // LISTA DE PELÍCULAS: Aquí es donde vos agregás tus favoritas
        const peliculas = [
            { titulo: "Esperando la Carroza", año: 1985, genero: "Comedia / Argentina" },
            { titulo: "El Secreto de sus Ojos", año: 2009, genero: "Drama / Suspenso" },
            { titulo: "Pulp Fiction", año: 1994, genero: "Crimen / Culto" },
            { titulo: "Mad Max: Fury Road", año: 2015, genero: "Acción" },
            { titulo: "Parasite", año: 2019, genero: "Thriller / Social" },
            { titulo: "The Truman Show", año: 1998, genero: "Drama / Comedia" },
            { titulo: "Spider-Man: Into the Spider-Verse", año: 2018, genero: "Animación" }
        ];

        function generarPelicula() {
            const display = document.getElementById('movie-display');
            const placeholder = document.getElementById('placeholder');
            const movieInfo = document.getElementById('movie-info');
            const titleElement = document.getElementById('movie-title');
            const detailsElement = document.getElementById('movie-details');

            // Elegir una al azar
            const indice = Math.floor(Math.random() * peliculas.length);
            const peli = peliculas[indice];

            // Mostrar el resultado con un efecto simple
            placeholder.style.display = 'none';
            movieInfo.style.display = 'block';
            titleElement.innerText = peli.titulo;
            detailsElement.innerText = peli.año + " • " + peli.genero;
        }
    </script>
